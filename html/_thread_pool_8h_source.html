211 #ifdef _WIN32
212 unsigned __stdcall WorkerThread( LPVOID arguments )
213 #else
214 void* WorkerThread( void* arguments )
215 #endif
216 */
217 {
218 
219 
220 
221  ThreadPool<ThreadInputType, ThreadOutputType> *threadPool = (ThreadPool<ThreadInputType, ThreadOutputType>*) arguments;
222 
223 
224  bool returnOutput;
225  ThreadOutputType (*userCallback)(ThreadInputType, bool *, void*);
226  ThreadInputType inputData;
227  ThreadOutputType callbackOutput;
228 
229  userCallback=0;
230 
231  void *perThreadData;
232  if (threadPool->perThreadDataFactory)
233  perThreadData=threadPool->perThreadDataFactory();
234  else if (threadPool->threadDataInterface)
235  perThreadData=threadPool->threadDataInterface->PerThreadFactory(threadPool->tdiContext);
236  else
237  perThreadData=0;
238 
239  // Increase numThreadsRunning
240  threadPool->numThreadsRunningMutex.Lock();
241  ++threadPool->numThreadsRunning;
242  threadPool->numThreadsRunningMutex.Unlock();
243 
244  while (1)
245  {
246 //#ifdef _WIN32
247  if (userCallback==0)
248  {
249  threadPool->quitAndIncomingDataEvents.WaitOnEvent(1000);
250  }
251 // #else
252 // if (userCallback==0)
253 // RakSleep(30);
254 // #endif
255 
256  threadPool->runThreadsMutex.Lock();
257  if (threadPool->runThreads==false)
258  {
259  threadPool->runThreadsMutex.Unlock();
260  break;
261  }
262  threadPool->runThreadsMutex.Unlock();
263 
264  threadPool->workingThreadCountMutex.Lock();
265  ++threadPool->numThreadsWorking;
266  threadPool->workingThreadCountMutex.Unlock();
267 
268  // Read input data
269  userCallback=0;
270  threadPool->inputQueueMutex.Lock();
271  if (threadPool->inputFunctionQueue.Size())
272  {
273  userCallback=threadPool->inputFunctionQueue.Pop();
274  inputData=threadPool->inputQueue.Pop();
275  }
276  threadPool->inputQueueMutex.Unlock();
277 
278  if (userCallback)
279  {
280  callbackOutput=userCallback(inputData, &returnOutput,perThreadData);
281  if (returnOutput)
282  {
283  threadPool->outputQueueMutex.Lock();
284  threadPool->outputQueue.Push(callbackOutput, _FILE_AND_LINE_ );
285  threadPool->outputQueueMutex.Unlock();
286  }
287  }
288 
289  threadPool->workingThreadCountMutex.Lock();
290  --threadPool->numThreadsWorking;
291  threadPool->workingThreadCountMutex.Unlock();
292  }
293 
294  // Decrease numThreadsRunning
295  threadPool->numThreadsRunningMutex.Lock();
296  --threadPool->numThreadsRunning;
297  threadPool->numThreadsRunningMutex.Unlock();
298 
299  if (threadPool->perThreadDataDestructor)
300  threadPool->perThreadDataDestructor(perThreadData);
301  else if (threadPool->threadDataInterface)
302  threadPool->threadDataInterface->PerThreadDestructor(perThreadData, threadPool->tdiContext);
303 
304 
305 
306 
307  return 0;
