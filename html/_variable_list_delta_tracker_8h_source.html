38  bool WriteVar(const VarType &varData)
39  {
40  RakNet::BitStream temp;
41  temp.Write(varData);
42  if (nextWriteIndex>=variableList.Size())
43  {
44  variableList.Push(VariableLastValueNode(temp.GetData(),temp.GetNumberOfBytesUsed()),_FILE_AND_LINE_);
45  nextWriteIndex++;
46  return true; // Different because it's new
47  }
48 
49  if (temp.GetNumberOfBytesUsed()!=variableList[nextWriteIndex].byteLength)
50  {
51  variableList[nextWriteIndex].lastData=(char*) rakRealloc_Ex(variableList[nextWriteIndex].lastData, temp.GetNumberOfBytesUsed(),_FILE_AND_LINE_);
52  variableList[nextWriteIndex].byteLength=temp.GetNumberOfBytesUsed();
53  memcpy(variableList[nextWriteIndex].lastData,temp.GetData(),temp.GetNumberOfBytesUsed());
54  nextWriteIndex++;
55  variableList[nextWriteIndex].isDirty=false;
56  return true; // Different because the serialized size is different
57  }
58  if (variableList[nextWriteIndex].isDirty==false && memcmp(temp.GetData(),variableList[nextWriteIndex].lastData, variableList[nextWriteIndex].byteLength)==0)
59  {
60  nextWriteIndex++;
61  return false; // Same because not dirty and memcmp is the same
62  }
63 
64  variableList[nextWriteIndex].isDirty=false;
65  memcpy(variableList[nextWriteIndex].lastData,temp.GetData(),temp.GetNumberOfBytesUsed());
66  nextWriteIndex++;
67  return true; // Different because dirty or memcmp was different
68  }
