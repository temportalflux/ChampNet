159  template <class VarType>
160  void SerializeVariable(SerializationContext *context, const VarType &variable)
161  {
162  if (context->newSystemSend)
163  {
164  if (context->variableHistory->variableListDeltaTracker.IsPastEndOfList()==false)
165  {
166  // previously sent data to another system
167  context->bitStream->Write(true);
168  context->bitStream->Write(variable);
169  context->anyVariablesWritten=true;
170  }
171  else
172  {
173  // never sent data to another system
174  context->variableHistory->variableListDeltaTracker.WriteVarToBitstream(variable, context->bitStream);
175  context->anyVariablesWritten=true;
176  }
177  }
178  else if (context->serializationMode==UNRELIABLE_WITH_ACK_RECEIPT)
179  {
180  context->anyVariablesWritten|=
181  context->variableHistory->variableListDeltaTracker.WriteVarToBitstream(variable, context->bitStream, context->changedVariables->bitField, context->changedVariables->bitWriteIndex++);
182  }
183  else
184  {
185  if (context->variableHistoryIdentical)
186  {
187  // Identical serialization to a number of systems
188  if (didComparisonThisTick==false)
189  context->anyVariablesWritten|=
190  context->variableHistory->variableListDeltaTracker.WriteVarToBitstream(variable, context->bitStream);
191  // Else bitstream is written to at the end
192  }
193  else
194  {
195  // Per-system serialization
196  context->anyVariablesWritten|=
197  context->variableHistory->variableListDeltaTracker.WriteVarToBitstream(variable, context->bitStream);
198  }
199  }
200  }
