225  template <class node_type, class weight_type, bool allow_unlinkedNodes>
226  bool WeightedGraph<node_type, weight_type, allow_unlinkedNodes>::GetShortestPath(DataStructures::List<node_type> &path, node_type startNode, node_type endNode, weight_type INFINITE_WEIGHT)
227  {
228  path.Clear(false, _FILE_AND_LINE_);
229  if (startNode==endNode)
230  {
231  path.Insert(startNode, _FILE_AND_LINE_);
232  path.Insert(endNode, _FILE_AND_LINE_);
233  return true;
234  }
235 
236  if (isValidPath==false || rootNode!=startNode)
237  {
238  ClearDijkstra();
239  GenerateDisjktraMatrix(startNode, INFINITE_WEIGHT);
240  }
241 
242  // return the results
243  bool objectExists;
244  unsigned col,row;
245  weight_type currentWeight;
246  DataStructures::Queue<node_type> outputQueue;
247  col=costMatrixIndices.GetIndexFromKey(endNode, &objectExists);
248  if (costMatrixIndices.Size()<2)
249  {
250  return false;
251  }
252  if (objectExists==false)
253  {
254  return false;
255  }
256  node_type vertex;
257  row=costMatrixIndices.Size()-2;
258  if (row==0)
259  {
260  path.Insert(startNode, _FILE_AND_LINE_);
261  path.Insert(endNode, _FILE_AND_LINE_);
262  return true;
263  }
264  currentWeight=costMatrix[row*adjacencyLists.Size() + col];
265  if (currentWeight==INFINITE_WEIGHT)
266  {
267  // No path
268  return true;
269  }
270  vertex=endNode;
271  outputQueue.PushAtHead(vertex, 0, _FILE_AND_LINE_);
272  row--;
273 #ifdef _MSC_VER
274 #pragma warning( disable : 4127 ) // warning C4127: conditional expression is constant
275 #endif
276  while (1)
277  {
278  while (costMatrix[row*adjacencyLists.Size() + col] == currentWeight)
279  {
280  if (row==0)
281  {
282  path.Insert(startNode, _FILE_AND_LINE_);
283  for (col=0; outputQueue.Size(); col++)
284  path.Insert(outputQueue.Pop(), _FILE_AND_LINE_);
285  return true;
286  }
287  --row;
288  }
289 
290  vertex=leastNodeArray[row];
291  outputQueue.PushAtHead(vertex, 0, _FILE_AND_LINE_);
292  if (row==0)
293  break;
294  col=costMatrixIndices.GetIndexFromKey(vertex, &objectExists);
295  currentWeight=costMatrix[row*adjacencyLists.Size() + col];
296  }
297 
298  path.Insert(startNode, _FILE_AND_LINE_);
299  for (col=0; outputQueue.Size(); col++)
300  path.Insert(outputQueue.Pop(), _FILE_AND_LINE_);
301  return true;
302  }

329  template <class node_type, class weight_type, bool allow_unlinkedNodes>
330  bool WeightedGraph<node_type, weight_type, allow_unlinkedNodes>::GetSpanningTree(DataStructures::Tree<node_type> &outTree, DataStructures::List<node_type> *inputNodes, node_type startNode, weight_type INFINITE_WEIGHT )
331  {
332  // Find the shortest path from the start node to each of the input nodes. Add this path to a new WeightedGraph if the result is reachable
333  DataStructures::List<node_type> path;
334  DataStructures::WeightedGraph<node_type, weight_type, allow_unlinkedNodes> outGraph;
335  bool res;
336  unsigned i,j;
337  for (i=0; i < inputNodes->Size(); i++)
338  {
339  res=GetShortestPath(path, startNode, (*inputNodes)[i], INFINITE_WEIGHT);
340  if (res && path.Size()>0)
341  {
342  for (j=0; j < path.Size()-1; j++)
343  {
344  // Don't bother looking up the weight
345  outGraph.AddConnection(path[j], path[j+1], INFINITE_WEIGHT);
346  }
347  }
348  }
349 
350  // Copy the graph to a tree.
351  DataStructures::Queue<NodeAndParent> nodesToProcess;
352  DataStructures::Tree<node_type> *current;
353  DataStructures::Map<node_type, weight_type> *adjacencyList;
354  node_type key;
355  NodeAndParent nap, nap2;
356  outTree.DeleteDecendants();
357  outTree.data=startNode;
358  current=&outTree;
359  if (outGraph.adjacencyLists.Has(startNode)==false)
360  return false;
361  adjacencyList = outGraph.adjacencyLists.Get(startNode);
362 
363  for (i=0; i < adjacencyList->Size(); i++)
364  {
365  nap2.node=RakNet::OP_NEW<DataStructures::Tree<node_type> >( _FILE_AND_LINE_ );
366  nap2.node->data=adjacencyList->GetKeyAtIndex(i);
367  nap2.parent=current;
368  nodesToProcess.Push(nap2, _FILE_AND_LINE_ );
369  current->children.Insert(nap2.node, _FILE_AND_LINE_);
370  }
371 
372  while (nodesToProcess.Size())
373  {
374  nap=nodesToProcess.Pop();
375  current=nap.node;
376  adjacencyList = outGraph.adjacencyLists.Get(nap.node->data);
377 
378  for (i=0; i < adjacencyList->Size(); i++)
379  {
380  key=adjacencyList->GetKeyAtIndex(i);
381  if (key!=nap.parent->data)
382  {
383  nap2.node=RakNet::OP_NEW<DataStructures::Tree<node_type> >( _FILE_AND_LINE_ );
384  nap2.node->data=key;
385  nap2.parent=current;
386  nodesToProcess.Push(nap2, _FILE_AND_LINE_ );
387  current->children.Insert(nap2.node, _FILE_AND_LINE_);
388  }
389  }
390  }
391 
392  return true;
393  }

396  void WeightedGraph<node_type, weight_type, allow_unlinkedNodes>::GenerateDisjktraMatrix(node_type startNode, weight_type INFINITE_WEIGHT)
397  {
398  if (adjacencyLists.Size()==0)
399  return;
400 
401  costMatrix = RakNet::OP_NEW_ARRAY<weight_type>(adjacencyLists.Size() * adjacencyLists.Size(), _FILE_AND_LINE_ );

406  node_type adjacentKey;
407  unsigned adjacentIndex;

409  DataStructures::Map<node_type, weight_type> *adjacencyList;
410  DataStructures::Heap<weight_type, node_type, false> minHeap;
411  DataStructures::Map<node_type, weight_type> openSet;
412 
413  for (col=0; col < adjacencyLists.Size(); col++)
414  {
415  // This should be already sorted, so it's a bit inefficient to do an insertion sort, but what the heck
416  costMatrixIndices.Insert(adjacencyLists.GetKeyAtIndex(col),adjacencyLists.GetKeyAtIndex(col), true, _FILE_AND_LINE_);
417  }
418  for (col=0; col < adjacencyLists.Size() * adjacencyLists.Size(); col++)
419  costMatrix[col]=INFINITE_WEIGHT;
420  currentNode=startNode;
421  row=0;
422  currentNodeWeight=0;
423  rootNode=startNode;
424 
425  // Clear the starting node column
426  if (adjacencyLists.Size())
427  {
428  adjacentIndex=adjacencyLists.GetIndexAtKey(startNode);
429  for (row2=0; row2 < adjacencyLists.Size(); row2++)
430  costMatrix[row2*adjacencyLists.Size() + adjacentIndex]=0;
431  }
432 
433  while (row < adjacencyLists.Size()-1)
434  {
435  adjacencyList = adjacencyLists.Get(currentNode);
436  // Go through all connections from the current node. If the new weight is less than the current weight, then update that weight.
437  for (col=0; col < adjacencyList->Size(); col++)
438  {
439  edgeWeight=(*adjacencyList)[col];
440  adjacentKey=adjacencyList->GetKeyAtIndex(col);
441  adjacentIndex=adjacencyLists.GetIndexAtKey(adjacentKey);
442  adjacentNodeWeight=costMatrix[row*adjacencyLists.Size() + adjacentIndex];
443 
444  if (currentNodeWeight + edgeWeight < adjacentNodeWeight)
445  {
446  // Update the weight for the adjacent node
447  for (row2=row; row2 < adjacencyLists.Size(); row2++)
448  costMatrix[row2*adjacencyLists.Size() + adjacentIndex]=currentNodeWeight + edgeWeight;
449  openSet.Set(adjacentKey, currentNodeWeight + edgeWeight);
450  }
451  }
452 
453  // Find the lowest in the open set
454  minHeap.Clear(true,_FILE_AND_LINE_);
455  for (openSetIndex=0; openSetIndex < openSet.Size(); openSetIndex++)
456  minHeap.Push(openSet[openSetIndex], openSet.GetKeyAtIndex(openSetIndex),_FILE_AND_LINE_);

461  {
462  for (j=0; j < adjacencyLists.Size(); j++)
463  {
464  RAKNET_DEBUG_PRINTF("%2i ", costMatrix[i*adjacencyLists.Size() + j]);

476  }
477

479  leastNodeArray[row]=minHeap.Pop(0);
480  currentNode=leastNodeArray[row];
481  openSet.Delete(currentNode);

483  }

489  {
